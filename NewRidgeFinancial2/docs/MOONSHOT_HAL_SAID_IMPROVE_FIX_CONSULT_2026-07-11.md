# apex_denial_workflow_pack.py
from datetime import datetime, timedelta
from typing import Any
from nr2_local_db import insert_office_task, query_office_tasks

# Pull denial codes from payer_reference so we don’t hardcode
from payer_reference_store import load_payer_reference

def _denial_triggers() -> list[str]:
    payers = load_payer_reference().get("payers", [])
    codes = set()
    for p in payers:
        for c in (p.get("commonDenialCodes") or []):
            codes.add(str(c).upper())
    return list(codes)

def auto_create_denial_tasks(era_claims: list[dict], dry_run: bool = False) -> list[dict]:
    """
    Called after ERA 835 ingest. Creates office_tasks assignee='Steve' for denials.
    Idempotent: checks existing tasks by claim_id + denial_code.
    """
    created = []
    triggers = _denial_triggers()
    for claim in era_claims:
        claim_id = str(claim.get("claim_id") or claim.get("patient_control_number") or "")
        for adj in claim.get("adjustments", []):
            code = str(adj.get("code") or "").upper()
            # Match CO-45, PR-119, etc.
            if any(code.startswith(t.split("-")[0]) and code.endswith(t.split("-")[1]) for t in triggers if "-" in t):
                # Check dup
                existing = query_office_tasks(claim_id=claim_id, category="denial_follow_up")
                if any(str(e.get("denial_code")) == code for e in existing):
                    continue
                task: dict[str, Any] = {
                    "title": f"Denial follow-up: {code}",
                    "category": "denial_follow_up",
                    "assignee": "Steve",  # Office Manager per hal-memo-index
                    "due_date": (datetime.now() + timedelta(days=3)).isoformat(),
                    "claim_id": claim_id,
                    "denial_code": code,
                    "payer_name": claim.get("payer_name"),
                    "amount": adj.get("amount"),
                    "created_at": datetime.now().isoformat(),
                    "updated_at": datetime.now().isoformat(),
                    "done": 0
                }
                if not dry_run:
                    insert_office_task(task)
                created.append(task)
    return created
```
**Validation:** Import test ERA with CO-45; verify task appears in `office_tasks` with `assignee='Steve'`; verify no duplicate on re-import.

from datetime import datetime
from nr2_local_db import connect

def record_eob_received(claim_id: str, era_date: str, payer: str):
    with connect() as conn:
        conn.execute("""
            INSERT INTO eob_backlog (claim_id, received_date, payer, posted)
            VALUES (?, ?, ?, 0)
            ON CONFLICT(claim_id) DO UPDATE SET received_date=excluded.received_date
        """, (claim_id, era_date, payer))
        conn.commit()

def get_backlog(days: int = 3) -> list[dict]:
    """Returns EOBs received but not posted within threshold."""
    with connect() as conn:
        rows = conn.execute("""
            SELECT claim_id, received_date, payer, 
                   julianday('now') - julianday(received_date) as age_days
            FROM eob_backlog 
            WHERE posted = 0 AND age_days > ?
            ORDER BY age_days DESC
        """, (days,)).fetchall()
    return [dict(r) for r in rows]
```

from payer_reference_store import search_payers

def normalize_softdent_label(sd_label: str) -> dict:
    """
    Maps raw SoftDent carrier string to canonical payer reference entry.
    Returns {'canonical_id': 'metlife-dental', 'tesiaPayerId': '0000E', ...} or 
            {'canonical_id': 'unknown', 'raw': sd_label}
    """
    hits = search_payers(sd_label, limit=1)
    if hits:
        return {
            "canonical_id": hits[0]["id"],
            "name": hits[0]["name"],
            "tesiaPayerId": hits[0].get("tesiaPayerId"),
            "eligibilityNotes": hits[0].get("eligibilityNotes"),
            "matched": True
        }
    return {"canonical_id": "unknown", "raw": sd_label, "matched": False, "needs_mapping": True}
